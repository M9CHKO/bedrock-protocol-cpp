Example

#include <bedrock/bedrock.hpp>
#include <iostream>

int main() {
    auto client = bedrock::createClient({
        .host = "localhost",
        .port = 19132,
        .username = "Bot",
        .version = "1.20.40",
        .offline = true
    });

    client.on("start_game", [](const bedrock::Packet&) {
        std::cout << "Joined world\n";
    });

    return client.run();
}
